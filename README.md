angular-d3-charts
=================

D3 snippets wrapped for use with AngularJS


USAGE

  <div donut-progress="sample"></div>



  // Donut Progress Chart

  $scope.sample = { 
    data: some_value,
    max: some_max,
    showPercentage: false,
    arcWidth: 20,
    transition: '',
    duration: 500,
    newChart: true
  };
